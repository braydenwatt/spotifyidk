import java.util.ArrayList;

public class Playlist {
    private String title;

    public SpotifyUser getOwner() {
        return owner;
    }

    private SpotifyUser owner;
    private ArrayList<Song> songs;

    public Playlist(String title, SpotifyUser owner, ArrayList<Song> song) {
        this.title = title;
        this.owner = owner;
        this.songs = song;
    }
    public String getTitle() {
        return title;
    }
    public void addSong(Song song) {
        this.songs.add(song);
    }
    public boolean containsSong(Song song) {
        return this.songs.contains(song);
    }
    public boolean removeSong(Song song) {
        System.out.println(this.songs.remove(song));
        return this.songs.remove(song);
    }
    public Song playRandomSong() {
        return this.songs.get((int) (Math.random() * this.songs.size()));
    }
    public String toString() {
        return "Title: " + this.title + "\n Owner: " + this.owner.getUsername() + "\n Songs: " + this.songs.toString();
    }
}
