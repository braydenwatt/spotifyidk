import java.util.ArrayList;

public class SpotifyUser {
    private String username;
    private ArrayList<Playlist> ownedPlaylists;
    private ArrayList<Playlist> followedPlaylists;

    public SpotifyUser(String username, ArrayList<Playlist> ownedPlaylists, ArrayList<Playlist> followedPlaylists) {
        this.username = username;
        this.ownedPlaylists = ownedPlaylists;
        this.followedPlaylists = followedPlaylists;
    }
    public String toString() {
        String output = "Username: " + this.username + "\n";
        output += "Owned Playlists: " + "\n";
        for(Playlist playlist : this.ownedPlaylists) {
            output += playlist.getTitle();
        }
        output += "\n";
        output += "Followed Playlists: " + "\n";
        for(Playlist playlist : this.followedPlaylists) {
            output += playlist.getTitle();
        }
        return output;
    }
    public void addPlaylistToOwned(Playlist playlist) {
        this.ownedPlaylists.add(playlist);
    }
    public Playlist createPlaylist(String title, Song firstSong) {
        for(Playlist playlist : this.ownedPlaylists) {
            System.out.println(playlist.getTitle());
            if(playlist.getTitle().equals(title)){
                return null;
            }
        }
        Playlist playlist = new Playlist(title, this, new ArrayList<Song>());
        playlist.addSong(firstSong);
        return playlist;
    }
    public Playlist getOwnedPlaylist(String playlistTitle) {
        for(Playlist playlist : this.ownedPlaylists) {
            if (playlist.getTitle().equals(playlistTitle)) {
                return playlist;
            }
        }
        return null;
    }
    public Playlist deleteOwnedPlaylist(String playlistTitle) {
        this.ownedPlaylists.remove(this.getOwnedPlaylist(playlistTitle));
        return null;
    }
    //based on PLAYLIST or playlist title
    private boolean playlistExists(Playlist playlist) {
        for (Playlist ownedPlaylist : this.ownedPlaylists) {
            if (ownedPlaylist.getTitle().equals(playlist.getTitle())) {
                return true;
            }
        }
        return false;
    }
    private boolean playlistExistsFollowed(Playlist playlist) {
        for (Playlist followedPlaytlist : this.followedPlaylists) {
            if (followedPlaytlist.getTitle().equals(playlist.getTitle())) {
                return true;
            }
        }
        return false;
    }
    private boolean playlistContainsSong(Playlist playlist, Song song) {
       if(playlist.containsSong(song)){
           return true;
       }
        return false;
    }

    public boolean addToOwnedPlaylist(Playlist playlist, Song song) {
        if(!(playlist.getOwner().getUsername().equals(this.getUsername()))){
            System.out.println(this.username + " does not own the playlist");
            return false;
        }
        if (!playlistExists(playlist)) {
            System.out.println("Playlist does not exist in owned");
            return false;
        }
        if (playlistContainsSong(playlist, song)) {
            System.out.println("Playlist already contains song");
            return false;
        }
    playlist.addSong(song);
    return true;
    }
    public boolean removeFromOwnedPlaylist(Playlist playlist, Song song) {
        if(!(playlist.getOwner().getUsername().equals(this.getUsername()))){
            System.out.println(this.username + " does not own the playlist");
            return false;
        }
        if (!playlistExists(playlist)) {
            System.out.println("Playlist does not exist in owned");
            return false;
        }

        if (playlistContainsSong(playlist, song)) {
            System.out.println("Playlist already contains song");
            return false;
        }
        playlist.removeSong(song);
        return true;
    }
    public boolean addToFollowedPlaylist(Playlist playlist, Song song) {
        if(!(playlist.getOwner().getUsername().equals(this.getUsername()))){
            System.out.println(this.username + " does not own the playlist");
            return false;
        }
        if (!playlistExistsFollowed(playlist)) {
            System.out.println("Playlist does not exist in followed");
            return false;
        }

        if (playlistContainsSong(playlist, song)) {
            System.out.println("Playlist already contains song");
            return false;
        }
        playlist.addSong(song);
        return true;
    }
    public boolean removeFromFollowedPlaylist(Playlist playlist, Song song) {
        if(!(playlist.getOwner().getUsername().equals(this.getUsername()))){
            System.out.println(this.username + " does not own the playlist");
            return false;
        }
        if (!playlistExistsFollowed(playlist)) {
            System.out.println("Playlist does not exist in followed");
            return false;
        }

        if (playlistContainsSong(playlist, song)) {
            System.out.println("Playlist already contains song");
            return false;
        }
        playlist.removeSong(song);
        return true;
    }
    public String getUsername(){
        return this.username;
    }
}
